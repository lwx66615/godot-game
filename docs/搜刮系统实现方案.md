## 一、需求分析

### 1.1 设计文档关键信息

| 设计点 | 说明 |
|--------|------|
| 时间消耗 | 搜刮/翻找消耗15分钟（游戏内时间） |
| 体力消耗 | 5%/次 |
| 物品刷新 | 不刷新，搜刮完即无 |
| 物品分布 | 按房间分布，不同位置有不同物品 |

### 1.2 核心功能

1. 玩家靠近可搜刮物体 → 显示交互提示
2. 按E搜刮 → 消耗时间和体力
3. 获得物品（或空） → 物品进入物品栏
4. 已搜刮物体标记 → 不能重复搜刮

## 三、详细设计

### 3.1 Searchable 类

```gdscript
extends Area2D
class_name Searchable

@export var search_name: String = "储物柜"
@export var loot_table: Array[Dictionary] = []
@export var search_time: float = 15.0  # 秒（对应游戏内15分钟）
@export var stamina_cost: float = 5.0  # 百分比

var is_searched: bool = false

signal searched(searchable: Searchable, items: Array)

func _ready():
    add_to_group("searchable")
    collision_layer = 4  # interactable层
    collision_mask = 1   # 检测player层
    # 添加碰撞形状
    body_entered.connect(_on_body_entered)
    body_exited.connect(_on_body_exited)

func search() -> bool:
    if is_searched:
        return false
    
    # 检查体力
    var player: Player = _get_player()
    if player and player.stamina < stamina_cost:
        push_warning("体力不足")
        return false
    
    # 消耗体力
    if player:
        player.stamina -= stamina_cost
        player.stamina_changed.emit(player.stamina)
    
    # 消耗时间
    if GameManager.instance:
        GameManager.instance.consume_time(int(search_time))
    
    # 生成掉落
    var items: Array = _generate_loot()
    
    # 添加到物品栏
    for item in items:
        Inventory.instance.add_item(item["id"], item["count"])
    
    is_searched = true
    searched.emit(self, items)
    
    return true

func _generate_loot() -> Array:
    var items: Array = []
    for loot in loot_table:
        if randf() < loot.get("chance", 1.0):
            items.append({
                "id": loot["id"],
                "count": loot.get("count", 1)
            })
    return items
```

### 3.2 GameUI 集成

在 `game_ui.gd` 中添加：

```gdscript
var nearby_searchable = null

# 在 _input 中添加搜刮交互
elif nearby_searchable and not is_in_dialogue:
    nearby_searchable.search()

# 添加静态方法
static func set_nearby_searchable(searchable, is_nearby: bool):
    if instance:
        if is_nearby:
            instance.nearby_searchable = searchable
        elif instance.nearby_searchable == searchable:
            instance.nearby_searchable = null
```
