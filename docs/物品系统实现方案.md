## 四、物品栏设计

### 4.1 数据结构

```gdscript
# 普通物品栏（12格）
# 每格: {"id": "item_id", "count": 1} 或 {"id": "", "count": 0}（空）
var items: Array[Dictionary] = []

# 关键道具栏（5格）
# 每格: "item_id" 或 ""（空）
var key_items: Array[String] = []

# 已获得文档ID列表
var documents: Array[String] = []

const MAX_ITEMS := 12
const MAX_KEY_ITEMS := 5
```

### 4.2 核心方法

```gdscript
# 添加物品 -> bool
func add_item(item_id: String, count: int = 1) -> bool:
    # 1. 获取物品数据
    var item_data := get_item_data(item_id)
    if item_data.is_empty():
        return false
    
    # 2. 关键道具 → 放入关键道具栏
    if item_data.category == ItemCategory.KEY_ITEM:
        return _add_key_item(item_id)
    
    # 3. 文档 → 放入文档列表
    if item_data.category == ItemCategory.DOCUMENT:
        return _add_document(item_id)
    
    # 4. 尝试堆叠到现有槽位
    if item_data.max_stack > 1:
        for slot in items:
            if slot.id == item_id and slot.count < item_data.max_stack:
                var space := item_data.max_stack - slot.count
                var add := mini(count, space)
                slot.count += add
                count -= add
                if count <= 0:
                    emit_signal("inventory_changed")
                    return true
    
    # 5. 寻找空槽位
    while count > 0:
        var empty_index := _find_empty_slot()
        if empty_index == -1:
            emit_signal("inventory_changed")
            return count < original_count  # 部分成功
        
        var add := mini(count, item_data.max_stack)
        items[empty_index] = {"id": item_id, "count": add}
        count -= add
    
    emit_signal("inventory_changed")
    return true

# 移除物品 -> bool
func remove_item(item_id: String, count: int = 1) -> bool:
    # 查找并减少数量
    for i in range(items.size()):
        if items[i].id == item_id:
            if items[i].count >= count:
                items[i].count -= count
                if items[i].count <= 0:
                    items[i] = {"id": "", "count": 0}
                emit_signal("inventory_changed")
                return true
            else:
                count -= items[i].count
                items[i] = {"id": "", "count": 0}
    
    emit_signal("inventory_changed")
    return count <= 0

# 使用物品 -> bool
func use_item(slot_index: int) -> bool:
    if slot_index < 0 or slot_index >= items.size():
        return false
    
    var slot := items[slot_index]
    if slot.id == "" or slot.count <= 0:
        return false
    
    # 检查使用限制
    if not _can_use_item():
        return false
    
    # 应用效果
    _apply_effects(slot.id)
    
    # 消耗时间
    var item_data := get_item_data(slot.id)
    if GameManager.instance:
        GameManager.instance.consume_time(GameManager.TIME_USE_ITEM)
    
    # 消耗品减少数量
    if item_data.category == ItemCategory.CONSUMABLE or \
       item_data.category == ItemCategory.ESCAPE_ITEM:
        slot.count -= 1
        if slot.count <= 0:
            slot = {"id": "", "count": 0}
    
    emit_signal("inventory_changed")
    return true
```

## 五、使用效果实现

### 5.1 效果应用函数

```gdscript
func _apply_effects(item_id: String) -> void:
    var item_data := get_item_data(item_id)
    var player := _get_player()
    var ghost := _get_ghost()
    
    for effect in item_data.effects:
        match effect.type:
            EffectType.RESTORE_HUNGER:
                if player:
                    player.hunger = mini(100.0, player.hunger + effect.value)
            EffectType.RESTORE_THIRST:
                if player:
                    player.thirst = mini(100.0, player.thirst + effect.value)
            EffectType.RESTORE_STAMINA:
                if player:
                    player.stamina = mini(100.0, player.stamina + effect.value)
            EffectType.RESTORE_SANITY:
                if player:
                    player.sanity = mini(100.0, player.sanity + effect.value)
            EffectType.COST_SANITY:
                if player:
                    player.sanity = maxi(0.0, player.sanity - effect.value)
            EffectType.HEAL_LIGHT:
                if player:
                    player.heal_light_wound()
            EffectType.HEAL_HEAVY:
                if player:
                    player.heal_heavy_wound()
            EffectType.ATTRACT_GHOST:
                if ghost:
                    ghost.attract_to_player(effect.value)
            EffectType.HIDE_FROM_GHOST:
                if ghost:
                    ghost.set_perception_zero(effect.value)
```

### 5.2 使用限制检查

```gdscript
func _can_use_item() -> bool:
    var player := _get_player()
    
    # 奔跑中不能使用物品
    if player and player.is_running:
        return false
    
    # 躲藏中不能使用物品（会暴露位置）
    if player and player.is_hiding():
        return false
    
    return true
```

## 六、物品拾取点

### 6.1 ItemPickup 类

```gdscript
class_name ItemPickup
extends Area2D

@export var item_id: String = ""
@export var count: int = 1

signal item_picked_up(item_id: String, count: int)

func _ready():
    add_to_group("item_pickup")
    collision_layer = 4  # interactable层
    collision_mask = 1   # 检测player层
    body_entered.connect(_on_body_entered)
    body_exited.connect(_on_body_exited)

func _on_body_entered(body: Node2D):
    if body is Player:
        var item_name := Inventory.get_item_name(item_id)
        GameUI.show_interact_prompt("按E拾取 " + item_name)
        GameUI.set_nearby_item(self, true)

func _on_body_exited(body: Node2D):
    if body is Player:
        GameUI.hide_interact_prompt()
        GameUI.set_nearby_item(self, false)

func pickup() -> bool:
    if Inventory.add_item(item_id, count):
        item_picked_up.emit(item_id, count)
        queue_free()
        return true
    return false
```
